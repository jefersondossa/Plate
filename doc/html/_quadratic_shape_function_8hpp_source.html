67 // for triangles and tetrahedrons
68 template<>
69 void QuadShapeFunction<2>::evaluate(const Coords& xi, Values& phi) const {
70 
71  const double xsi1 = xi(0);
72  const double xsi2 = xi(1);
73  const double xsi3 = 1. - xsi1 - xsi2;
74 
75  phi(0) = xsi3 * (2.0 * xsi3 - 1.0);
76  phi(1) = xsi1 * (2.0 * xsi1 - 1.0);
77  phi(2) = xsi2 * (2.0 * xsi2 - 1.0);
78  phi(3) = 4.0 * xsi3 * xsi1;
79  phi(4) = 4.0 * xsi1 * xsi2;
80  phi(5) = 4.0 * xsi2 * xsi3;
81 
82  // element conectivity
83  // 2
84  // 54
85  // 031
86 
87  return;
88 }

92 
93  const double xsi1 = xi(0);
94  const double xsi2 = xi(1);
95  const double xsi3 = xi(2);
96 
97  phi(0) = 2.0 * (xsi1 - 0.50) * xsi1;
98  phi(1) = 2.0 * (xsi2 - 0.50) * xsi2;
99  phi(2) = 2.0 * (xsi3 - 0.50) * xsi3;
100  phi(3) = (2.0 - 2.0 * xsi3 - 2.0 * xsi2 - 2.0 * xsi1 - 1.0) \
101  * (1.0 - xsi1 - xsi2 - xsi3);
102  phi(4) = 4.0 * xsi1 * xsi2;
103  phi(5) = 4.0 * xsi1 * xsi3;
104  phi(6) = 4.0 * xsi1 * (1.0 - xsi1 - xsi2 - xsi3);
105  phi(7) = 4.0 * xsi2 * xsi3;
106  phi(8) = 4.0 * xsi3 * (1.0 - xsi1 - xsi2 - xsi3);
107  phi(9) = 4.0 * xsi2 * (1.0 - xsi1 - xsi2 - xsi3);
108 
109  // element conectivity
110  //layer 1
111  // 3
112  // 8 9
113  // 2 7 1
114  //layer 2

116  // 54
117  //layer 3
118  // 0

134  dphi(0,1) = 4. * xsi1 - 1.;

136

141  dphi(1,0) = -4. * xsi3 + 1.;

143  dphi(0,4) = 4. * xsi2;
144  dphi(1,4) = 4. * xsi1;

148 
149  dphi(0,3) = 4. * (xsi3 - xsi1);
150  dphi(1,3) = -4. * xsi1;
