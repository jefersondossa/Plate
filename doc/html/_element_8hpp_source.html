323 template<>
324 void Element<2>::getJacobianMatrix(ublas::bounded_vector<double,2>& xsi) {
325 
326  //Computes the spatial Jacobian matrix and its inverse
327 
328  typename QuadShapeFunction<2>::ValueDeriv dphi;
329 
330  shapeQuad.evaluateGradient(xsi,dphi);
331 
332  double dx_dxsi1 = 0.;
333  double dx_dxsi2 = 0.;
334  double dy_dxsi1 = 0.;
335  double dy_dxsi2 = 0.;
336 
337  for (int i=0; i<6; i++){
338  dx_dxsi1 += localNodes_(i,0) * dphi(0,i);
339  dx_dxsi2 += localNodes_(i,0) * dphi(1,i);
340  dy_dxsi1 += localNodes_(i,1) * dphi(0,i);
341  dy_dxsi2 += localNodes_(i,1) * dphi(1,i);
342  };
343 
344  //Computing the jacobian determinant
345  djac_ = dx_dxsi1 * dy_dxsi2 - dx_dxsi2 * dy_dxsi1;
346 
347  //Computing Jacobian inverse
348  ainv_(0,0) = dy_dxsi2 / djac_;
349  ainv_(0,1) = -dy_dxsi1 / djac_;
350  ainv_(1,0) = -dx_dxsi2 / djac_;
351  ainv_(1,1) = dx_dxsi1 / djac_;
352 
353  return;
354 
355 };

359 
360  //Computes the spatial Jacobian matrix and its inverse
361 
362  typename QuadShapeFunction<3>::ValueDeriv dphi;
363 
364  shapeQuad.evaluateGradient(xsi,dphi);
365 
366  double dx_dxsi1 = 0.;
367  double dx_dxsi2 = 0.;
368  double dx_dxsi3 = 0.;
369  double dy_dxsi1 = 0.;
370  double dy_dxsi2 = 0.;
371  double dy_dxsi3 = 0.;
372  double dz_dxsi1 = 0.;
373  double dz_dxsi2 = 0.;
374  double dz_dxsi3 = 0.;
375 
376  for (int i=0; i<10; i++){
377  dx_dxsi1 += localNodes_(i,0) * dphi(0,i);
378  dx_dxsi2 += localNodes_(i,0) * dphi(1,i);
379  dx_dxsi3 += localNodes_(i,0) * dphi(2,i);
380  dy_dxsi1 += localNodes_(i,1) * dphi(0,i);
381  dy_dxsi2 += localNodes_(i,1) * dphi(1,i);
382  dy_dxsi3 += localNodes_(i,1) * dphi(2,i);
383  dz_dxsi1 += localNodes_(i,2) * dphi(0,i);
384  dz_dxsi2 += localNodes_(i,2) * dphi(1,i);
385  dz_dxsi3 += localNodes_(i,2) * dphi(2,i);
386  };
387 
388  //Computing the jacobian determinant
389  djac_ = dx_dxsi1 * dy_dxsi2 * dz_dxsi3 - dx_dxsi1 * dy_dxsi3 * dz_dxsi2
390  + dx_dxsi2 * dy_dxsi3 * dz_dxsi1 - dx_dxsi2 * dy_dxsi1 * dz_dxsi3
391  + dx_dxsi3 * dy_dxsi1 * dz_dxsi2 - dx_dxsi3 * dy_dxsi2 * dz_dxsi1;
392 
393  //Computing Jacobian inverse
394  ainv_(0,0) = (dy_dxsi2 * dz_dxsi3 - dy_dxsi3 * dz_dxsi2) / djac_;
395  ainv_(0,1) = (dx_dxsi3 * dz_dxsi2 - dx_dxsi2 * dz_dxsi3) / djac_;
396  ainv_(0,2) = (dx_dxsi2 * dy_dxsi3 - dx_dxsi3 * dy_dxsi2) / djac_;
397  ainv_(1,0) = (dy_dxsi3 * dz_dxsi1 - dy_dxsi1 * dz_dxsi3) / djac_;
398  ainv_(1,1) = (dx_dxsi1 * dz_dxsi3 - dx_dxsi3 * dz_dxsi1) / djac_;
399  ainv_(1,2) = (dx_dxsi3 * dy_dxsi1 - dx_dxsi1 * dy_dxsi3) / djac_;
400  ainv_(2,0) = (dy_dxsi1 * dz_dxsi2 - dy_dxsi2 * dz_dxsi1) / djac_;
401  ainv_(2,1) = (dx_dxsi2 * dz_dxsi1 - dx_dxsi1 * dz_dxsi2) / djac_;
402  ainv_(2,2) = (dx_dxsi1 * dy_dxsi2 - dx_dxsi2 * dy_dxsi1) / djac_;

490 template<>
491 void Element<2>::getElemStiffMatrix(){
492 
493  stiffnessMatrix.clear();
494 
495  ublas::bounded_matrix<double, 3, 3 > hooke;
496 
497  hooke.clear();
498 
499  // For EPT
500  double k = elastic_ / (1. - poisson_ * poisson_);
501  hooke(0,0) = k;
502  hooke(0,1) = k * poisson_;
503  hooke(1,0) = k * poisson_;
504  hooke(1,1) = k;
505  hooke(2,2) = k * (1. - poisson_) * 0.5;
506 
507  // For EPD
508  // double k = elastic_ / ((1. + poisson_) * (1. - 2. * poisson_));
509  // hooke(0,0) = k * (1. - poisson_);
510  // hooke(0,1) = k * poisson_;
511  // hooke(1,0) = k * poisson_;
512  // hooke(1,1) = k * (1. - poisson);
513  // hooke(2,2) = k * (1. - 2. * poisson_) * 0.5;
514 
515  typename QuadShapeFunction<2>::Coords xsi;
516  int index = 0;
517 
518  for(typename NormalQuad::QuadratureListIt it = nQuad.begin();
519  it != nQuad.end(); it++){
520 
521  ublas::bounded_matrix<double, 3,12> bMatrix, aux;
522  bMatrix.clear(); aux.clear();
523 
524  //Defines the integration points adimentional coordinates
525  xsi(0) = nQuad.PointList(index,0);
526  xsi(1) = nQuad.PointList(index,1);
527 
528  //Computes the shape functions
529  shapeQuad.evaluate(xsi,phi_);
530 
531  //Returns the quadrature integration weight
532  weight_ = nQuad.WeightList(index);
533 
534  //Computes the jacobian matrix
535  getJacobianMatrix(xsi);
536 
537  //Computes spatial derivatives
538  getSpatialDerivatives(xsi);
539 
540  for (int i = 0; i < 6; i++){
541  bMatrix(0,2*i ) = dphi_dx(0,i);
542  bMatrix(2,2*i ) = dphi_dx(1,i);
543 
544  bMatrix(1,2*i+1) = dphi_dx(1,i);
545  bMatrix(2,2*i+1) = dphi_dx(0,i);
546  };
547 
548  aux = prod(hooke,bMatrix);
549 
550  stiffnessMatrix += prod(trans(bMatrix),aux) * djac_ * weight_ * thick_;
551 
552  index++;
553  };
554 
555  return;
